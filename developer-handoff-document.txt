// services/ai/formGenerator.js
class AIFormGenerator {
async generateForm(userPrompt) {
const intent = await this.parseUserIntent(userPrompt);
const structure = await this.generateFormStructure(intent);
const validated = this.validateOutput(structure);
if (!validated.isValid) {
return this.refineAmbiguities(validated.issues);
}
return validated.form;
}
}
